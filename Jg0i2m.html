문제

사용자가 입력한 단어에서 어근과 접미사를 구분하는 문제입니다 이를 해결하는 다음 프로그램의 실행상태를 확인하세요 
-ly, -ment, -ance로 끝나는 단어를 입력하세요: lovely movement
lovely = love-ly
movement = move-ment

프로그램 코드

	import java.util.Scanner;
		
	public class Suffix
	{
1		public static void main( String[] args ) {		
			Scanner scan = new Scanner( System.in );
	
2			System.out.print( "-ly, -ment, -ance로 끝나는 단어를 입력하세요: " );
3			String input = scan.nextLine();
	
4			String[] suffixes = { "ly", "ment", "ance" };
5			String[] words = input.split( " " );
6			for( String word : words ) {
7				System.out.print( word + " = " );
	
8				String root = word;
9				for( String suffix : suffixes )
10					root = root.replace( suffix, "" );
11				System.out.print( root );
				
12				for( String suffix : suffixes ) {
13					if ( word.contains( suffix ) ) {
14						int index = word.indexOf( suffix );
15						System.out.println( "-" + word.substring( index ) );
16						break;
					}
				}
			}
17		}
	}
